# Example 4: Using a Loop to Make API Calls for Each Prompt
import requests

prompts = [
    "Generate an image of a futuristic city.",
    "Generate an image of a serene landscape.",
    "Generate an image of a bustling market."
]

api_url = "https://api.example.com/generate"
api_key = "your_api_key_here"

for prompt in prompts:
    response = requests.post(api_url, headers={"Authorization": f"Bearer {api_key}"}, json={"prompt": prompt, "num_images": 1})
    if response.status_code == 200:
        print(f"Prompt: {prompt}, Generated Images: {response.json()}")
    else:
        print(f"Failed to generate images for prompt: {prompt}, Error: {response.text}")

# Example 5: Using a Function to Generate an Image
import requests

def generate_image(prompt, api_url, api_key):
    response = requests.post(api_url, headers={"Authorization": f"Bearer {api_key}"}, json={"prompt": prompt, "num_images": 1})
    if response.status_code == 200:
        return response.json()
    else:
        return f"Error: {response.text}"

api_url = "https://api.example.com/generate"
api_key = "your_api_key_here"
prompt = "Generate an image of a futuristic city."

generated_image = generate_image(prompt, api_url, api_key)
print(f"Generated Image: {generated_image}")

# Example 6: Using args in a Function for GenAI
def generate_images(api_url, api_key, *prompts):
    for prompt in prompts:
        response = requests.post(api_url, headers={"Authorization": f"Bearer {api_key}"}, json={"prompt": prompt, "num_images": 1})
        if response.status_code == 200:
            print(f"Prompt: {prompt}, Generated Images: {response.json()}")
        else:
            print(f"Failed to generate images for prompt: {prompt}, Error: {response.text}")

api_url = "https://api.example.com/generate"
api_key = "your_api_key_here"

generate_images(api_url, api_key, "Generate an image of a futuristic city.", "Generate an image of a serene landscape.", "Generate an image of a bustling market.")

# Example 7: Using kwargs in a Function for GenAI (Revised)
def generate_images_with_kwargs(api_url, api_key, **kwargs):
    valid_keys = {"prompt", "num_images", "style"}
    options = {k: v for k, v in kwargs.items() if k in valid_keys}
    
    response = requests.post(api_url, headers={"Authorization": f"Bearer {api_key}"}, json=options)
    if response.status_code == 200:
        print(f"Options: {options}, Generated Images: {response.json()}")
    else:
        print(f"Failed to generate images with options: {options}, Error: {response.text}")

api_url = "https://api.example.com/generate"
api_key = "your_api_key_here"

generate_images_with_kwargs(api_url, api_key, prompt="Generate an image of a futuristic city.", num_images=2, style="cyberpunk")
